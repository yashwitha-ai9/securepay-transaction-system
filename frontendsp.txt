        // Function to update mobile input based on country code
        function updateMobileInput() {
            const countryCode = document.getElementById('country-code');
            const mobileInput = document.getElementById('mobile');
            const selectedOption = countryCode.options[countryCode.selectedIndex];
            const maxLength = selectedOption.getAttribute('data-length');
            
            // Update maxlength and placeholder
            mobileInput.maxLength = maxLength;
            mobileInput.placeholder = Enter ${maxLength}-digit mobile number;
            
            // Remove non-numeric characters and limit length
            mobileInput.value = mobileInput.value.replace(/[^0-9]/g, '').slice(0, maxLength);
        }
        
        // Function to initialize mobile input
        function initializeMobileInput() {
            const countryCode = document.getElementById('country-code');
            const mobileInput = document.getElementById('mobile');
            
            if (countryCode && mobileInput) {
                // Set initial placeholder
                updateMobileInput();
                
                // Update when country code changes
                countryCode.addEventListener('change', updateMobileInput);
            }
        }

        // Function to toggle speaker
        function toggleSpeaker() {
            speakerEnabled = !speakerEnabled;
            
            // Update all speaker icons
            const speakerButtons = document.querySelectorAll('.floating-speaker');
            speakerButtons.forEach(button => {
                const icon = button.querySelector('i');
                if (speakerEnabled) {
                    icon.className = 'fas fa-volume-up';
                    button.style.backgroundColor = '#7C3AED';
                    button.style.color = 'white';
                } else {
                    icon.className = 'fas fa-volume-mute';
                    button.style.backgroundColor = '#EF4444';
                    button.style.color = 'white';
                    // Stop any ongoing speech
                    if ('speechSynthesis' in window) {
                        window.speechSynthesis.cancel();
                    }
                }
            });
            
            // Speak status change in selected language
            if (speakerEnabled) {
                speakText(getLocalizedText('voiceEnabled'));
            }
        }
        
        // Function to show a specific page

        // Function to select language
        function selectLanguage(element, language) {
            // Remove selected class from all language cards
            document.querySelectorAll('.language-card').forEach(card => {
                card.classList.remove('selected');
            });
            
            // Add selected class to clicked card
            element.classList.add('selected');

        // Text-to-speech functionality
        function speakText(text) {
            if (!speakerEnabled) return;
            
            // Check if browser supports speech synthesis
            if ('speechSynthesis' in window) {
                // Cancel any ongoing speech
                window.speechSynthesis.cancel();
                
                // Create a new speech synthesis utterance
                const utterance = new SpeechSynthesisUtterance(text);
                
                // Set properties based on selected language
                utterance.lang = window.speechLanguageCode || 'en-US';
                utterance.rate = 0.9;
                utterance.pitch = 1.0;
                utterance.volume = 1.0;
                
                // Speak the text
                window.speechSynthesis.speak(utterance);
            }
        }

        // Function to verify UPI PIN during payment
        function verifyUPIPin() {
            const verifyPinInputs = document.querySelectorAll('#verify-pin-page .pin-input');
            const enteredPin = Array.from(verifyPinInputs).map(input => input.value).join('');
            
            if (enteredPin.length !== 4) {

                if (speakerEnabled) speakText('Please enter your complete 4-digit UPI PIN');
                return;
            }
            
            if (enteredPin !== userUPIPin) {

                // Clear the PIN inputs
                verifyPinInputs.forEach(input => input.value = '');
                verifyPinInputs[0].focus();
                return;

            // PIN is correct, proceed to fingerprint authentication
            showPage('fingerprint-auth-page');
        }

        // Main validation function
        function validateAndContinue(currentPage, nextPage) {
            let isValid = false;
            let errorMessage = '';
            
            switch(currentPage) {
                case 'language-page':
                    isValid = window.selectedLanguage;
                    errorMessage = 'Please select a language to continue.';
                    break;
                    
                case 'mobile-page':
                    const mobile = document.getElementById('mobile').value.trim();
                    const countryCode = document.getElementById('country-code').value;
                    const checkbox = document.querySelector('input[type="checkbox"]').checked;
                    const selectedOption = document.getElementById('country-code').options[document.getElementById('country-code').selectedIndex];
                    const requiredLength = parseInt(selectedOption.getAttribute('data-length'));
                    
                    isValid = mobile && mobile.length === requiredLength && /^\d+$/.test(mobile) && checkbox;
                    errorMessage = Please enter a valid ${requiredLength}-digit mobile number and accept terms and conditions.;
                    if (isValid) {
                        userData.mobile = mobile;
                        userData.countryCode = countryCode;
                        // Update OTP page with mobile number
                        document.querySelector('#otp-page p').textContent = Enter the 4-digit code sent to ${countryCode} ${mobile};
                    }
                    break;
                    
                case 'otp-page':
                    const otpInputs = document.querySelectorAll('.otp-input');
                    isValid = Array.from(otpInputs).every(input => input.value.trim().length === 1);
                    errorMessage = 'Please enter the complete 4-digit OTP to verify your mobile number.';
                    break;
                    
                case 'bank-selection-page':
                    isValid = window.selectedBank;
                    errorMessage = 'Please select your bank to continue with account setup.';
                    if (isValid) {
                        userData.selectedBank = window.selectedBank;
                    }
                    break;
                    
                case 'account-details-page':
                    const accountNumber = document.getElementById('account-number').value.trim();
                    const confirmAccount = document.getElementById('confirm-account').value.trim();
                    const ifscCode = document.getElementById('ifsc-code').value.trim();
                    const accountHolder = document.getElementById('account-holder').value.trim();
                    
                    isValid = accountNumber && confirmAccount && ifscCode && accountHolder && 
                             accountNumber === confirmAccount && accountNumber.length >= 9 && ifscCode.length >= 11;
                    errorMessage = 'Please fill all account details correctly. Account numbers must match and IFSC code must be valid.';
                    
                    // Store user data and update throughout app
                    if (isValid) {
                        userData.accountHolderName = accountHolder;
                        userData.accountNumber = accountNumber;
                        userData.ifscCode = ifscCode;
                        updateUserDataThroughoutApp();
                    }
                    break;
                    
                case 'upi-pin-page':
                    const pinInputs = document.querySelectorAll('.pin-input');
                    const allPinInputs = Array.from(pinInputs);
                    const firstFourPins = allPinInputs.slice(0, 4);
                    const lastFourPins = allPinInputs.slice(4, 8);
                    
                    const pin = firstFourPins.map(input => input.value).join('');
                    const confirmPin = lastFourPins.map(input => input.value).join('');
                    
                    isValid = pin.length === 4 && confirmPin.length === 4 && pin === confirmPin && /^\d{4}$/.test(pin);
                    errorMessage = 'Please enter and confirm your 4-digit UPI PIN. Both PINs must match and contain only numbers.';
                    
                    if (isValid) {
                        userUPIPin = pin; // Store the PIN for later validation
                    }
                    break;
                    
                default:
                    isValid = true;
            }
            
            if (!isValid) {
                alert(errorMessage);
                if (speakerEnabled) speakText(errorMessage);
                return;
            }
            
            showPage(nextPage);
        }
